011    public class CharacteresList
012    {
013        // variables d'instance - remplacez l'exemple qui suit par le vôtre
014        private HashMap<String, Characteres> aCharacteresList;;
015    
016        /**
017         * Constructeur d'objets de classe CharacterList
018         */
019        public CharacteresList()
020        {
021            this.aCharacteresList = new HashMap<String, Characteres>();
022        }  
023        
024        public void addCharacteres(final String pNom, final Characteres pCha)
025        {
026            this.aCharacteresList.put(pNom, pCha);
027        }
028        
029        public boolean isEmpty()
030        {
031            return this.aCharacteresList.isEmpty();
032        }
033        
034        public Characteres get(final String pN)
035        {
036            return this.aCharacteresList.get(pN);
037        }
038        
039        public HashMap<String, Characteres> getCharacteresList()
040        {
041            return this.aCharacteresList;
042        }
043        
044        public boolean hasCharacteres(final String pC)
045        {
046            return this.aCharacteresList.containsKey(pC);
047        }
048        
049        public String getCharactereString()
050        {
051            String RS = "\n";
052            Set<String> keySet = aCharacteresList.keySet();
053            for (String vCha : keySet) RS = RS + aCharacteresList.get(vCha).getNom() + " : " + aCharacteresList.get(vCha).getChaDescription() + "\n";
054            return RS;
055        }
056    }
