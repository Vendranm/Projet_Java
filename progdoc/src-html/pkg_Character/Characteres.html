010    public class Characteres

015        private  String aDialogues;
016        private  static ItemList aItems;
017        private  Room aCurrentRoom;

019        {
020            this.aNom = pNom;
021            this.aDescription = pDesc;

025        }

034            return this.aDescription;
035        }
036

038        {
039            String vDesc = (aNom + (" ") + aDescription);
040            return vDesc;
041        }
042        
043        public  String getDialogue()
044        {
045            return aDialogues;
046        }
047        
048        public static ItemList getChaInventaire()
049        {
050            return aItems;
051        }
052        
053        public  void removeChaInventaire(final String pI)
054        {
055            aItems.remove(pI);
056        }
057        
058        public  Room getCurrentRoom()
