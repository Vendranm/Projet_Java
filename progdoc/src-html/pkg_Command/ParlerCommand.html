013    public class ParlerCommand extends Command
014    {
015        // variables d'instance - remplacez l'exemple qui suit par le vôtre
016        private CommandWords aCommandWords;
017        private int Choix;
018        /**
019         * Constructeur d'objets de classe ParlerCommand
020         */
021        public ParlerCommand()
022        {
023        }
024        
025        @Override
026        public void execute(final Player Joueur)
027        {
028            if (! hasSecondWord()) 
029            {
030                GameEngine.getGui().println("A qui voulez-vous parlez ?"); 
031                return;
032            }
033            
034            String vCommand = getSecondWord();
035            Characteres vCha = Joueur.getCurrentRoom().getChaList().get(vCommand);
036            
037            if (vCha == null) GameEngine.getGui().println("La personne n'est pas dans la pièce.");
038            else
039            {
040                String vCommand2 = getSecondWord();
041                Item vChaItem = Joueur.getCurrentRoom().getChaItemList().get(vCommand2);
042                GameEngine.getGui().println(vCha.getDialogue());
043                GameEngine.getGui().println("Choix de discussion : \n");
044                GameEngine.getGui().println("1-Vous etes tres jolie puis-je vous offrire un verre ?");
045                GameEngine.getGui().println("2-Tu as de très beau yeaux...");
046                GameEngine.getGui().println(3 + "- y");
047                GameEngine.getGui().println(4 + "- u");
048                GameEngine.getGui().println("Que voulez-vous dire");
049    //             GameEngine.getGui().scanf("%d", Choix); 
050                switch (Choix)
051                {
052                    case 1 : 
053                    vCha.removeChaInventaire(vCommand2);
054                    Joueur.putInventaire(vChaItem);
055    
056                    break;
057                    
058                    case 2 :
059                    vCha.removeChaInventaire(vCommand2);
060                    Joueur.putInventaire(vChaItem);
061                    GameEngine.getGui().println("Vous avez gagner le numéro de" + vCha.getNom());
062                    break;
063                    
064                    case 3 :
065                    vCha.removeChaInventaire(vCommand2);
066                    Joueur.putInventaire(vChaItem);
067                    GameEngine.getGui().println("Vous avez gagner le numéro de" + vCha.getNom());
068                    break;
069                    
070                    case 4 :
071                    GameEngine.getGui().println("Vous vous etes pris un basche");
072                    break;
073                }
074            }
075        }
076    }
