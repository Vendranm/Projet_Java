001    package pkg_Command;
002    import pkg_Character.Player;
003    import pkg_Mechanics.GameEngine;
004    import pkg_Item.Item;
005    /**
006     * Implementation of the 'quit' user command.
007     * 
008     * @author Michael Kolling and David J. Barnes
009     * @version 2011.07.31
010     */
011    public class BoireCommand extends Command
012    {
013        private CommandWords commandWords;
014        /**
015         * Constructor for objects of class QuitCommand
016         */
017        public BoireCommand()
018        {
019        }
020    
021        /**
022         * "Quit" was entered. Check the argument to see whether
023         * we really quit the game. Return true, if we should quit, false
024         * otherwise.
025         */
026        @Override
027        public void execute(Player Joueur)
028        {
029            if(!hasSecondWord()) 
030            {
031                GameEngine.getGui().println("Boire quoi?");
032                return;
033            }
034    
035            String name = getSecondWord();
036            Item item = Joueur.getInventaire().get(name);
037    
038            if (item == null)
039            {
040                GameEngine.getGui().println("Vous ne possedez pas " + name);
041            }
042    
043            if (!item.isEdible())
044            {
045                GameEngine.getGui().println("Vous ne pouvez pas boire " + name);
046            }
047            
048            else
049            {
050                if (name.equals("MagicBeer")) 
051                {
052                    GameEngine.setLimite(GameEngine.getLimite() + 5*60);
053                    GameEngine.getGui().println("Vous avez gagn√© 5 minutes");
054                }
055                           
056                Joueur.getInventaire().remove(name);
057                GameEngine.getGui().println("Vous avez bu " + name);
058            }
059        }//boire()
060    }
