024        @Override
025        public void execute(Player Joueur)
026        {
027             if (! hasSecondWord()) 
028            {
029                GameEngine.getGui().println("\n Quel objet voulez-vous prendre ?\n"); 
030            }
031    
032            String vCommand2 = getSecondWord();
033            Item vItemRoom = Joueur.getCurrentRoom().getItemList().get(vCommand2);
034    
035            if (vItemRoom == null) 
036            {
037                GameEngine.getGui().println("\n Cet objet ne figure pas dans la pièce\n");
038            }
039            else 
040            { 
041                Joueur.putInventaire(vItemRoom);
042                Joueur.getCurrentRoom().getItemList().remove(vCommand2);
043                GameEngine.getGui().println("\n Vous avez ramassé " + vItemRoom.getName() + ".\n" );
044            }
045        }
