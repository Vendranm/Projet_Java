023    
024        @Override
025        public void execute(Player Joueur)
026        {
027            if(! hasSecondWord())
028            {
029                GameEngine.getGui().println("Quel item voulez-vous poser ?");
030            }
031    
032            String vCommand2 = getSecondWord();
033            Item vItemInv = Joueur.getInventaire().get(vCommand2);
034    
035            if (vItemInv == null) 
036            {
037                GameEngine.getGui().println("L'objet voulu n'est pas en votre posséssion ");
038            }
039            else
040            {
041                Joueur.removeInventaire(vCommand2);
042                Joueur.getCurrentRoom().getItemList().addItem(vCommand2, vItemInv);
043                GameEngine.getGui().println("Vous avez posé l'objet " + vItemInv.getName());
044            }
045        }
046    }
