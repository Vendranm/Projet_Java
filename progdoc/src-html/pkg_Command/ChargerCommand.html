025        @Override
026        public void execute(Player Joueur)
027        {
028            String vC = getSecondWord();
029           Item vBeamerI = Joueur.getInventaire().get(vC);
030           Beamer vBeamer = new Beamer("Beamer", "Cet objet vous permettra de revenir à une pièce.", 3, true);
031           if (! hasSecondWord()) 
032           {
033               GameEngine.getGui().println("Charger quoi ?");
034               return;
035           }
036           
037           if(! Joueur.getInventaire().hasItem("Beamer"))
038           {
039               GameEngine.getGui().println("Vous ne possedez pas de Beamer."); 
040           }
041           
042           try 
043           {
044               vBeamer = (Beamer) vBeamerI;
045           }
046           catch (java.lang.ClassCastException pE) 
047           {
048               GameEngine.getGui().println(" Je ne peux pas charger ça...\n");
049           }
050           
051           vBeamer.setSavedRoom(Joueur.getCurrentRoom());
052           GameEngine.getGui().println("Le Beamer est chargé.");
053           vBeamer.setCharged(true);
054        }
