001    package pkg_Command;
002    import pkg_Character.Player;
003    import pkg_Mechanics.GameEngine;
004    
005    /**
006     * Implementation of the 'quit' user command.
007     * 
008     * @author Michael Kolling and David J. Barnes
009     * @version 2011.07.31
010     */
011    public class QuitterCommand extends Command
012    {
013        private CommandWords commandWords;
014        /**
015         * Constructor for objects of class QuitCommand
016         */
017        public QuitterCommand()
018        {
019        }
020    
021        /**
022         * "Quit" was entered. Check the argument to see whether
023         * we really quit the game. Return true, if we should quit, false
024         * otherwise.
025         */
026        @Override
027        public void execute(final Player Joueur)
028        {
029            if(!hasSecondWord())
030            {
031                GameEngine.getGui().println("Quitter quoi?");
032            }
033            else
034            {
035                GameEngine.endGame();
036            }
037        }
038    
039    }
