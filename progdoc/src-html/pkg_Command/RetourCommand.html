024        @Override
025        public void execute(Player Joueur)
026        {
027            if (Joueur.getBackRoom().empty())
028            {
029                GameEngine.getGui().println("Vous etes deja a la piece de depart!");
030                return;
031            }
032    
033            else 
034            {
035              if (Joueur.getCurrentRoom().getShortDescription().equals("dans le bar le Beffroi")) 
036              {
037                  GameEngine.getGui().println("Vous êtes coincé. Il faut que vous trouviez la clef pour sortir.");
038              }
039              Joueur.backPlayer();
040              GameEngine.getGui().println(Joueur.getCurrentRoom().getLongDescription());
041              GameEngine.getGui().showImage(Joueur.getCurrentRoom().getImageName());
042            }
043        }
