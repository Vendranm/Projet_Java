001    package pkg_Command;
002    import java.util.StringTokenizer;
003    
004    /**
005     * This class is the main class of the "World of Zuul" application. 
006     * "World of Zuul" is a very simple, text based adventure game.  
007     *
008     * This parser reads user input and tries to interpret it as an "Adventure"
009     * command. Every time it is called it reads a line from the terminal and
010     * tries to interpret the line as a two word command. It returns the command
011     * as an object of class Command.
012     *
013     * The parser has a set of known command words. It checks user input against
014     * the known commands, and if the input is not one of the known commands, it
015     * returns a command object that is marked as an unknown command.
016     * 
017     * @author  Michael Kolling and David J. Barnes
018     * @version 2011.07.31
019     */
020    
021    public class Parser 
022    {
023    
024        private CommandWords aValidsCommand;  // (voir la classe CommandWords)
025    
026        /**
027         * Constructeur par defaut d'un Parser pour lire ce que l'on tape sur le terminal 
028         */
029        public Parser() 
030        {
031            this.aValidsCommand = new CommandWords();
032        } // Parser()
033    
034        /**
035         * @return The next command from the user.
036         */
037        public Command getCommand(final String inputLine) 
038        {
039            //         String vInputLine;    // contiendra toute la ligne tapee
040            String word1;
041            String word2;
042    
043            StringTokenizer tokenizer = new StringTokenizer(inputLine);
044    
045            if(tokenizer.hasMoreTokens())
046                word1 = tokenizer.nextToken();      // get first word
047            else
048                word1 = null;
049            if(tokenizer.hasMoreTokens())
050                word2 = tokenizer.nextToken();      // get second word
051            else
052                word2 = null;
053    
054            // note: we just ignore the rest of the input line.
055            // Now check whether this word is known. If so, create a command
056            // with it. If not, create a "null" command (for unknown command).
057            Command vCommand = this.aValidsCommand.getCommand(word1);
058            if(vCommand != null)
059            {
060                vCommand.setSecondWord(word2);
061            }
062            return vCommand;
063        } // getCommand()
064    
065        /**
066         * Affiche une liste de commandes valides.
067         */
068        public String showCommands ()
069        { 
070            return this.aValidsCommand.showAll();
071        }
072    }
