001    package pkg_Command;
002    import pkg_Mechanics.GameEngine;
003    import pkg_Character.Player;
004    import pkg_Room.Room;
005    
006    /**
007     * Implementation of the 'go' user command.
008     * 
009     * @author Michael Kolling and David J. Barnes
010     * @version 2011.07.31
011     */
012    public class AllerCommand extends Command
013    {
014        /**
015         * Constructor for objects of class GoCommand
016         */
017        public AllerCommand()
018        {
019        }
020    
021        /** 
022         * Try to go to one direction. If there is an exit, enter the new
023         * room, otherwise print an error message. Returns always 'false'.
024         */
025        @Override
026        public void execute(final Player Joueur)
027        {
028    
029            if(!hasSecondWord()) 
030            {
031                // if there is no second word, we don't know where to go...
032                GameEngine.getGui().println("Aller où?");
033                return;
034            }
035            else
036            {
037                String direction = getSecondWord();
038    
039                // Try to leave current room.
040                Room vPreviousRoom= Joueur.getCurrentRoom();
041                Joueur.getBackRoom().push(vPreviousRoom);
042                Room nextRoom = Joueur.getCurrentRoom().getExit(direction);
043                //System.out.println(acurrentRoom);
044                
045                if (nextRoom == null)
046                    GameEngine.getGui().println("Il n'y a pas d'issue!");
047    
048                if (Joueur.getCurrentRoom().getShortDescription().equals("dans le bar le Beffroi") && !Joueur.getInventaire().hasItem("Clef") && direction.equals("Rue"))
049                {
050                    GameEngine.getGui().println("La porte est bloqué vous ne pouvez sortir sans la clef.");
051                    return;
052                }
053                else 
054                {
055                    Joueur.setCurrentRoom ( nextRoom);
056                    GameEngine.getGui().println(Joueur.getCurrentRoom().getLongDescription());
057                    if(Joueur.getCurrentRoom().getImageName() != null)
058                        GameEngine.getGui().showImage(Joueur.getCurrentRoom().getImageName());
059                }
060            }
061        }
062    }
