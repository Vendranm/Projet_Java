001    import java.util.Set;
002    import java.util.HashMap;
003    import java.util.Iterator;
004    
005    /**
006     * Class Room - a room in an adventure game.
007     *
008     * This class is part of the "World of Zuul" application. 
009     * "World of Zuul" is a very simple, text based adventure game.  
010     *
011     * A "Room" represents one location in the scenery of the game.  It is 
012     * connected to other rooms via exits.  For each existing exit, the room 
013     * stores a reference to the neighboring room.
014     * 
015     * @author  Michael Kolling and David J. Barnes
016     * @version 1.0 (February 2002) DBMOD:04/04/2008
017     */
018    
019    public class Room 
020    {
021        private String description;
022        private HashMap<String,Room> exits;// stores exits of this room.
023        private HashMap<String,Item> aItems;
024        private String imageName;
025        public Room aBar1Exit;
026        public Room aBar2Exit;
027        public Room aBar3Exit;
028        public Room aBar4Exit;
029        public Room aBar5Exit;
030        public Room aBar6Exit;
031        public Room aBar7Exit;
032        public Room aBar8Exit;
033        public Room aBar9Exit;
034        public Room aBar10Exit;
035        /**
036         * Create a room described "description" with a given image. 
037         * Initially, it has no exits. "description" is something like 
038         * "in a kitchen" or "in an open court yard".
039         */
040        public Room(final String description,final String image,final Item pItem) 
041        {
042            this.description = description;
043            this.exits = new HashMap<String,Room>();
044            this.aItems = new HashMap<String,Item>();
045            this.imageName = image;
046        }//Room()
047    
048        /**
049         * Define an exit from this room.
050         */
051        public void setExit(final String direction,final Room neighbor) 
052        {
053            exits.put(direction, neighbor);
054        }//setExit()
055    
056        /**
057         * Return the description of the room (the one that was defined in the
058         * constructor).
059         */
060        public String getShortDescription()
061        {
062            return description;
063        }//getShortDescription()
064    
065        /**
066         * Return a long description of this room, in the form:
067         *     You are in the kitchen.
068         *     Exits: north west
069         */
070        public String getLongDescription()
071        {
072            String vString = "Vous Ãªtes " + this.description + ".\n" + this.getExitString() + "\n";
073            if (aItems != null) vString += this.getItemsString();
074            return vString;
075        }//getLongDescription()
076    
077        /**
078         * Return a string describing the room's exits, for example
079         * "Exits: north west".
080         */
081        private String getExitString()
082        {
083            StringBuilder returnString = new StringBuilder( "Exits: \n" );
084            for ( String vS : exits.keySet() )
085                returnString.append( " " + vS + "\n");
086            return returnString.toString();
087        }//getExitString()
088    
089        /**
090         * Return the room that is reached if we go from this room in direction
091         * "direction". If there is no room in that direction, return null.
092         */
093        public Room getExit(String direction) 
094        {
095            return exits.get(direction);
096        }//getExit()
097    
098        /**
099         * Return a string describing the room's image name
100         */
101        public String getImageName()
102        {
103            return imageName;
104        }//getImageName()
105        
106        public void addItem(final String pNom, final Item pItem)
107        {
108            this.aItems.put(pNom, pItem);
109        }
110        
111        public String getItemsString()
112        {
113            String RS = "Items : \n";
114            Set<String> keySet = aItems.keySet();
115            for (String vItem : keySet) RS = RS + aItems.get(vItem).getItemDescription() + "\n";
116            return RS;
117        }//getItem()
118        
119    }//class Room
